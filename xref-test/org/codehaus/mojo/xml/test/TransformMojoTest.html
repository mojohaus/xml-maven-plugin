57      public void runTestIt4( String dir, String targetFile )
58          throws Exception
59      {
60          TransformMojo mojo = (TransformMojo) newMojo( dir );
61          mojo.execute();
62          Document doc1 = parse( new File( dir, "xml/doc1.xml" ) );
63          doc1.normalize();
64          Document doc2 = parse( new File( dir, "target/generated-resources/xml/xslt/" + targetFile ) );
65          doc2.normalize();
66          Element doc1Element = doc1.getDocumentElement();
67          assertEquals( "doc1", doc1Element.getLocalName() );
68          assertNull( doc1Element.getNamespaceURI() );
69          Element doc2Element = doc2.getDocumentElement();
70          assertEquals( "doc2", doc2Element.getLocalName() );
71          assertNull( doc2Element.getNamespaceURI() );
72          Node text1 = doc1Element.getFirstChild();
73          assertNotNull( text1 );
74          assertNull( text1.getNextSibling() );
75          assertEquals( Node.TEXT_NODE, text1.getNodeType() );
76          Node text2 = doc2Element.getFirstChild();
77          assertNotNull( text2 );
78          assertNull( text2.getNextSibling() );
79          assertEquals( Node.TEXT_NODE, text2.getNodeType() );
80          assertEquals( text1.getNodeValue(), text2.getNodeValue() );
81      }

132     private String read( File file ) throws IOException
133     {
134         final StringBuffer sb = new StringBuffer();
135         final Reader reader = new InputStreamReader( new FileInputStream( file ), "UTF-8" );
136         final char[] buffer = new char[ 4096 ];
137         for ( ;; )
138         {
139             final int res = reader.read( buffer );
140             if ( res == -1 )
141             {
142                 break;
143             }
144             if ( res > 0 )
145             {
146                 sb.append( buffer, 0, res );
147             }
148         }
149         reader.close();
150         return sb.toString();
151     }

174     private String eval( Node contextNode, String str ) throws TransformerException, NoSuchMethodException, IllegalAccessException, InvocationTargetException
175     {
176         final String xsl = "<xsl:stylesheet version='1.0' xmlns:xsl='http://www.w3.org/1999/XSL/Transform'>\n"
177             + "<xsl:template match='*'>\n"
178             + "<xsl:value-of select='" + str + "'/>\n"
179             + "</xsl:template>\n"
180             + "</xsl:stylesheet>\n";
181         final StringWriter sw = new StringWriter();
182         final Transformer t = TransformMojo.newTransformerFactory( org.apache.xalan.processor.TransformerFactoryImpl.class.getName(), getClass().getClassLoader()).newTransformer( new StreamSource( new StringReader( xsl ) ) );
183         t.setOutputProperty( OutputKeys.OMIT_XML_DECLARATION, "yes" );
184         t.transform( new DOMSource( contextNode ), new StreamResult( sw ) );
185         return sw.toString();
186     }
